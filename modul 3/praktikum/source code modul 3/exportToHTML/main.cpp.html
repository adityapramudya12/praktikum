#include <iostream>
#include <string.h>// Menggunakan <cstring> untuk fungsi strrev dan strlen
using namespace std;

int main() {
    char tulisan1_07746[100], tulisan2_07746[100]; //(mendeklarasikan dua array karakter dengan nama tulisan1_07746 dan tulisan2_07746. Kapasitas masing-masing array adalah 100 elemen)
    cout << "Masukkan kalimat: ";
    cin.getline(tulisan1_07746, 100);//(inputan buat memasukkan kalimat sesuai apa yang ditunjukkan di soal)

    strcpy(tulisan2_07746, tulisan1_07746);//(menyalin isi dari tulisan1_07746 ke tulisan2_07746. Ini dilakukan untuk memiliki salinan kalimat yang nantinya akan diubah tanpa memengaruhi aslinya)
    strrev(tulisan1_07746);//(untuk membalikkan letak urutan pada string)

    cout << "Kalimat dibalik : " << tulisan1_07746 << endl;//(output dari strrev untuk membalikkan kalimat)
    cout << "Ejaan lama      : ";//(output dari strcpy untuk mengubah ejaan lama sesuai dengan for yang ada dibawah)

    //mengubah huruf ejaan baru ke ejaan lama
    for (int i_07746 = 0; i_07746 < strlen(tulisan2_07746); i_07746++) {
        if (tulisan2_07746[i_07746] == 'u') {
            cout << "oe";
        } else if (tulisan2_07746[i_07746] == 'J') { // Ubah 'J' menjadi 'j'
            cout << "DJ";
        } else if (tulisan2_07746[i_07746] == 'y') { // Ubah 'Y' menjadi 'y'
            cout << "j";
        } else if (tulisan2_07746[i_07746] == 'C') { // Ubah 'C' menjadi 'c'
            cout << "TJ";
        } else {
            cout << tulisan2_07746[i_07746];
        }
    }

    return 0;
}
